Code (1)

const int LEDPIN = 13;

void ledcontrolSetup() {
  pinMode(LEDPIN, OUTPUT);
}

void ledcontrolLedOn() {
  digitalWrite(LEDPIN, HIGH);
}

void ledcontrolLedOff() {
  digitalWrite(LEDPIN, LOW);
}

Code (2)

void period(int freq, int duty) {
  ledcontrolLedOn();
  delay(10 * duty / freq);
  ledcontrolLedOff();
  delay(10 * (100 - duty) / freq);
}

Code (3)

const int PERIOD = 1000;
const int FREQ = 50;

void setup() {
  ledcontrolSetup();
  Serial.begin(9600);
}

void loop() {
  for (int i = 10; i < 100; i += 10) {
    // must be a *long*!
    long start = millis();
    Serial.println(i);
    while (millis() - start < PERIOD) {
      period(FREQ, i);
    }
  }
}
