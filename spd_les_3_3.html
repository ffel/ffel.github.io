float calcParallelogram(float pTop, float pBottom, float pHeight) {
    float pWidth = (pTop + pBottom) / 2;
    float paralellogram = pWidth * pHeight;
    return paralellogram; 
}

Opgave 5.1 (2)

int somBovenOnder;
int bovenBreuk;
int oppervlakte;

void setup() {
  berekenOppervlakte( 20, 40, 10);
  println("Oppervlakte is: " + oppervlakte);
}

void berekenOppervlakte(int hoogte, int lengteBoven, int lengteOnder) {
  berekenSomBovenOnder( lengteBoven, lengteOnder );
  multiplyMetHoogte(hoogte);
  deelDoorTwee(); 
}

void berekenSomBovenOnder(int lengteBoven, int lengteOnder) {
  somBovenOnder = lengteBoven + lengteOnder;
}

void multiplyMetHoogte(int hoogte) {
  bovenBreuk = hoogte * somBovenOnder; 
}

void deelDoorTwee() {
  oppervlakte = bovenBreuk / 2;   
}

Afronding Moduleopgave 4

boolean lampenAan;

void setup() {
  size(500, 300);
}

void draw() {
  lampenAan = ! lampenAan;
  if (lampenAan) {
    tekenLampenAan();
  } else {
    tekenLampenUit();
  }
  delay (500);
}

void tekenLamp(int lampX, int kleur) {
  int lampY = 100;
  fill(#D3D3D3);
  rect(lampX - 25, lampY + 30, 50, 50);
  fill(kleur);
  ellipse(lampX, lampY, 100, 100);
}

void tekenLampenAan() {
  tekenLamp(width/4, #ff0000);
  tekenLamp(width/2, #00ff00);
  tekenLamp(width/4*3, #0000ff);
}

void tekenLampenUit() {
  tekenLamp(width/4, #550000);
  tekenLamp(width/2, #005500);
  tekenLamp(width/4*3, #000055);
}
