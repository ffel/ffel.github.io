Voorbeeld

//windowsize
int windowWidth = 0;
int windowW = constrain(windowWidth, 500, 1400);
int windowH = windowW / 3 * 2;

//numberbox
import controlP5.*;
ControlP5 theControl;

void setup() {
  size(400, 350);
  theControl = new ControlP5(this);
  createNumberBoxes(theControl);
  
  //grootte/kleur vlak
  size(windowW, windowH);
  background(0);
}

void draw() {
  println("lengte:" + leesLengte());
  println("gewicht: " + leesGewicht());
  
  //background resetten
  background(0);

  //BMI en van cm naar meter
  float Lengte = leesLengte() / 100;
  float BMI = leesGewicht() / (Lengte * Lengte);
  float scaleBMI = (windowH / 2) / 30 * BMI;
  
  //BMI displaybalk
  fill(255, 0, 255);
  rect(windowW / 2 - 50, windowH, 100, scaleBMI * -1);
  
  //categorielijnen
  fill(255, 255, 255);
  
  //obees
  rect(0, windowH / 2, windowW, 2);
  text("obees", 10, windowH / 2 - 10);
  
  //overgewicht 25/30 is de hoogte t.o.v. hoogte obees
  float linescaleover = windowH * 25 / 30 / 2;
  rect(0,windowH - linescaleover, windowW, 2);
  text("overgewicht", 10, windowH - linescaleover - 10);
  
  //normaal gewicht
  float linescalenormaal = windowH * 18.5 / 30 / 2;
  rect(0,windowH - linescalenormaal, windowW, 2);
  text("normaal", 10, windowH - linescalenormaal - 10);
  
  //ondergewicht
  text("ondergewicht", 10, windowH - linescalenormaal + 15);
}

import controlP5.*;
ControlP5 theControl;

int wWidth = 400;
int wHeight = 350;
int scale = 3;

void setup() {
  size(wWidth, wHeight);
  background(0);
  theControl = new ControlP5(this);
  createNumberBoxes(theControl);
}

void draw() {
  int BMI = round(leesGewicht() / pow(leesLengte(), 2) * 10000);
  println("lengte:" + leesLengte());
  println("gewicht: " + leesGewicht());
  println("BMI: " + BMI);
  
  //Overlappende zwarte balk
  fill(0);
  stroke(0);
  rect(wWidth / 2 - 50, 0, 100, wHeight - BMI - 1);
  
  //Lijnen met BMI waardes
  stroke(255);
  fill(255);
  text("Ondergewicht", 5, wHeight - 13.5 * scale - 1);
  line(0, wHeight - 18.5 * scale, wWidth, wHeight - 18.5 * scale);
  text("Normaal gewicht", 5, wHeight - 18.5 * scale - 1);
  line(0, wHeight - 25 * scale, wWidth, wHeight - 25 * scale);
  text("Overgewicht", 5, wHeight - 25 * scale - 1);
  line(0, wHeight - 30 * scale, wWidth, wHeight - 30 * scale);
  text("Obees", 5, wHeight - 30 * scale - 1);
  
  //Meter rect()
  fill(138, 22, 63);
  stroke(255);
  rect(wWidth / 2 - 50, wHeight, 100, - BMI * scale); 
}
