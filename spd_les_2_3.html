void keyReleased() {
  switch(keyCode) {
    case LEFT:
      posRobotX -= 20;
      break;
    case RIGHT:
      posRobotX += 20;
      break;
    case DOWN:
      posRobotY += 20;
      break;
    case UP:
      posRobotY -= 20;
      break;
    case ENTER:
      if(posRobotX == posVrachtX && posRobotY == posVrachtY) {
        isCarried = !isCarried;
      }
      break;
  }
}
