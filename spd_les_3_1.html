Opgave 4.2

Zet Hallo <naam>, hoe gaat het met je in vier kwadranten op het scherm

  • De functie text() mag maar één keer in de code voor komen.

Code

void setup() {
   size(600, 500); 
}

void draw() {
  groet("Harry", -1, -1);
  groet("Marianne", 1, -1);
  groet("Ludo", 1, 1);
  groet("Maaike", -1, 1);
}

void groet(String naam, int fx, int fy) {
  textAlign(CENTER, CENTER);
  String msg = "Hallo " + naam + ", hoe gaat het met je?";
  text(msg, width/2 + fx * width/4, height/2 + fy * height/4);
}

Opgave 4.3

Oppervlak parallelogram

\[O = (a + b)/2 * h\]

Code

int somBovenOnder;
int bovenBreuk;
int oppervlakte;

void setup() {
  berekenOppervlakte( 20, 40, 10);
  println("Oppervlakte is: " + oppervlakte);
}

void berekenOppervlakte(int hoogte, int lengteBoven, int lengteOnder) {
  berekenSomBovenOnder( lengteBoven, lengteOnder );
  multiplyMetHoogte(hoogte);
  deelDoorTwee(); 
}

void berekenSomBovenOnder(int lengteBoven, int lengteOnder) {
  somBovenOnder = lengteBoven + lengteOnder;
}

void multiplyMetHoogte(int hoogte) {
  bovenBreuk = hoogte * somBovenOnder; 
}

void deelDoorTwee() {
  oppervlakte = bovenBreuk / 2;   
}
