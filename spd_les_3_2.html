int wBreedte = 200;
int wHoogte = 200;
boolean lampKlik;
boolean knipper;

int lampSize = wBreedte/5;
void setup() {
  size(wBreedte, wHoogte);
  background(0);
  textAlign(CENTER);
}

void draw() {
  knop();
  if (lampKlik == true) {
    lampAan();
    if (knipper == true) {
      lampAan(); 
      knipper = !knipper;
    } else if (knipper == false) {
      lampUit();   
      knipper = !knipper;
    }
  } else if (lampKlik == false) {
    lampUit();
  }
  delay(500);
}

void lampAan() {
  fill(247, 7, 44);
  ellipse(1*wBreedte/4, wHoogte/3, lampSize, lampSize);
  fill(8, 252, 91);
  ellipse(2*wBreedte/4, wHoogte/3, lampSize, lampSize);
  fill(50, 130, 198);
  ellipse(3*wBreedte/4, wHoogte/3, lampSize, lampSize);
}

void lampUit() {
  fill(118, 7, 7);
  ellipse(1*wBreedte/4, wHoogte/3, lampSize, lampSize);
  fill(27, 134, 63);
  ellipse(2*wBreedte/4, wHoogte/3, lampSize, lampSize);
  fill(20, 65, 103);
  ellipse(3*wBreedte/4, wHoogte/3, lampSize, lampSize);
}

void knop() {
  fill(237, 237, 136);
  rect(wBreedte/10, wHoogte*0.75, lampSize*4, lampSize*0.75);
  fill(0);
  if (lampKlik == true) {
    text("Zet uit", wBreedte/2, wHoogte*0.75+wHoogte*0.08);
  } else {
    text("Zet Aan", wBreedte/2, wHoogte*0.75+wHoogte*0.08);
  }
}

void klik() {
  if (mouseX > wBreedte/10 && mouseX < (wBreedte/10)+(lampSize*4)
    && mouseY > wHoogte*0.75 && mouseY < (wHoogte*0.75)+(lampSize*0.75)) {
    lampKlik = !lampKlik;
  }
}

void mouseClicked() {
  klik();
}

boolean lampenAan;
boolean ingeschakeld;

int knopX, knopY, knopW, knopH;

void setup() {
  size(500, 300);

  textAlign(CENTER, CENTER);

  knopX = width / 4;
  knopY = height * 5 / 7;
  knopW = width / 2;
  knopH = height / 7;
}

void draw() {
  background(0);

  if (ingeschakeld) {
    lampenAan = ! lampenAan;
    if (lampenAan) {
      background(50);
      tekenLampenAan();
    } else {
      tekenLampenUit();
    }
    delay (250);
  } else {
    tekenLampenUit();
  }

  tekenKnop();
}

void mouseClicked() {
  if (mouseX > knopX && mouseX < knopX + knopW &&
    mouseY > knopY && mouseY < knopY + knopW) {
    ingeschakeld = !ingeschakeld;
  }
}

void tekenLamp(int lampX, int kleur) {
  int lampY = 100;
  fill(#D3D3D3);
  rect(lampX - 25, lampY + 30, 50, 50);
  fill(kleur);
  ellipse(lampX, lampY, 100, 100);
}

void tekenLampenAan() {
  tekenLamp(width/4, #ff0000);
  tekenLamp(width/2, #00ff00);
  tekenLamp(width/4*3, #0000ff);
}

void tekenLampenUit() {
  tekenLamp(width/4, #550000);
  tekenLamp(width/2, #005500);
  tekenLamp(width/4*3, #000055);
}

void tekenKnop() {
  fill(255);
  rect(knopX, knopY, knopW, knopH);
  fill(0);
  String label = "zet uit";
  if (ingeschakeld) {
    label = "zet aan";
  }
  text(label, knopX + knopW/2, knopY + knopH/2);
}

// Variabele 

int windowWidth = 500;
int windowHeight = 300;

boolean lampOut = false;
boolean blinking = false;


// Setup die opstart

void setup() {
  size(windowWidth, windowHeight);
  ellipseMode(CORNER);
}

// Draw die lampen tekent en laat knipperen 

void draw() {
  background(0);
  if (lampOut) {

    drawLamp(width/4, 50, #FF4646);
    drawLamp(200, 50, #64FF46);
    drawLamp(300, 50, #5654FC);
  } else {

    drawLamp(100, 50, #8E1313);
    drawLamp(200, 50, #117110);
    drawLamp(300, 50, #131190);
  }
  if (blinking) {
    lampOut = !lampOut;
    fill(#23ECF5);
    rect( windowWidth*0.20, windowHeight*0.75, windowWidth*0.60, windowHeight*0.20);
    textAlign(CENTER);
    textSize(40);
    fill(#EA551A);
    text("PRESS OFF ", windowWidth*0.5, windowHeight*0.9 );
  } else {
    lampOut = false;
    fill(#23ECF5);
    rect( windowWidth*0.20, windowHeight*0.75, windowWidth*0.60, windowHeight*0.20);
    textAlign(CENTER);
    textSize(40);
    fill(#EA551A);
    text("PRESS ON ", windowWidth*0.5, windowHeight*0.9 );
  }

  delay(150);
}

// Eigen methode voor lampen 

void drawLamp(int xPosition, int yPosition, int collor) {

  fill(100);
  rect(xPosition +20, yPosition + 70, 50, 50);
  fill(collor);
  ellipse(xPosition, yPosition, 90, 90);
}

// Mouseclick voor aan een uitzetten 

void mouseClicked() {
  if (mouseX >= width*0.2 && mouseX <= width * 0.8 && mouseY >= height*0.75 && mouseY <= height*0.95 ) { 
    blinking = !blinking;
  }
}

// Knop positie initialiseren
int butposx, butposy, butx, buty, textposx;
float textposy;

// Knop aan of uit
boolean knop = false;

// Stroom op de lampjes
boolean stroom = true;

// Standaardtekst op de knop
String tekst = "Zet aan";

void setup () {
  // Instellingen window
  size (600, 400);
  background (0);
  textSize (36);
  textAlign (CENTER);

  // Knop- en tekstpositie
  butposx = width/6;
  butposy = height/5*4;
  butx = width/6*4;
  buty = height/6;
  textposx = width / 2;
  textposy = height/5*4.5;
}

void draw () {
  // Maak het scherm volledig zwart
  rectMode (CORNER);
  fill (#000000);
  rect (0, 0, width, height);

  // Teken de knop en initialiseer de lampcontroller
  tekenknop ();
  lamphouders ();
  lampcontroller ();
}

void mouseClicked () {
  // Switch voor de knoptekst
  if (mouseX > butposx && mouseX < butposx + butx && mouseY > butposy && mouseY < butposy + buty) {
    if (knop == false) {
      tekst = "Zet uit";
    } else {
      tekst = "Zet aan";
    }
    knop = !knop;
  }
}

void tekenknop () {
  // Tekenen knop
  fill (#bbbbbb);
  rect (butposx, butposy, butx, buty);

  // Tekst op de knop schrijven
  fill (0);
  text (tekst, textposx, textposy);
  fill (#bbbbbb);
}

void lamphouders () {
  // Lamphouders tekenen
  rectMode (CENTER);
  rect (width/3-30, height/3+40, 80, 80);
  rect (width/2, height/3+40, 80, 80);
  rect ((width/3)*2+30, height/3+40, 80, 80);
}

void lampcontroller () {
  // Switch voor de stroom op de lampjes
  if (knop == true) {
    if (stroom == true) {
      // Lampjes knipperen aan
      lampaan ();
      delay (100);
    } else {
      // Lampjes knipperen uit
      lampuit ();
      delay (100);
    }
    // Switch voor het knipperen
    stroom = !stroom;
  } else {
    // Lampjes uit
    lampuit ();
  }
}

void lampaan () {
  // Lampjes aanzetten
  lamp (#ff0000, width/3-30);
  lamp (#00ff00, width/2);
  lamp (#0000ff, (width/3)*2+30);
}

void lampuit () {
  // Lampjes uitzetten
  lamp (#bb0000, width/3-30);
  lamp (#00bb00, width/2);
  lamp (#0000bb, (width/3)*2+30);
}

void lamp (int kleur, int xpositie) {
  // Lampjes tekenen en inkleuren
  fill (kleur);
  ellipse (xpositie, height/3, 120, 120);
}
