  int positie;
  float blokjeBreedte = breedte / nPosities;
  
  if ( mouseX < x ) {
    positie = 0;
  }
  else if (mouseX >= breedte + x ) {
    positie = nPosities - 1;

Nog Eentje Dan

float lengte = 180;
float gewicht = 80;
float cat_onder       = 18.5;
float cat_normal      = 25;
float cat_over        = 30;
float cat_obees       = 70;
float pixelsPerPoint;

int selectedSlider = 0;

void setup() {
  int breedte = 800;
  int hoogte = breedte / 16 * 9;
  size(breedte, hoogte);
  pixelsPerPoint = hoogte / cat_obees;
}

void draw() {
  fill(#3D3B3B);
  stroke(0);
  
  rect(0, 0, width - 1, height - 1);
  
  drawBMI();
  
  stroke(255);
  fill(255);
  
  drawCat("Ondergewicht", cat_onder);
  drawCat("Normaal", cat_normal);
  drawCat("Overgewicht", cat_over);
  drawCat("Obees", cat_obees);
  
  sliderFrame();
}

void drawCat(String text, float bmi){
  line(0, height - bmi * pixelsPerPoint, width, height - bmi * pixelsPerPoint);
  text(text, 10, height - bmi * pixelsPerPoint + 15);
}

void drawBMI(){
  float BMI = gewicht / pow(lengte / 100, 2);
  println("BMI: " + BMI);
  
  if(BMI <= cat_onder){
    fill(#FF0000);
  }else if(BMI < cat_normal){
    fill(#00FF00);
  }else if(BMI < cat_over){
    fill(#FFBF00);
  }else{
    fill(#FF0000);
  }
  

  rect(width / 2 - 100, height, 200, -BMI * pixelsPerPoint);
}

void sliderFrame(){
  if(mousePressed){
    switch(selectedSlider){
      case 0:
        if(mouseInSlider(width - 150, 50, 50, height - 100, lengte / 210)){
          selectedSlider = 1;
        }
        if(mouseInSlider(width -75, 50, 50, height - 100, gewicht / 140)){
          selectedSlider = 2;
        }
        break;
      case 1:
        lengte = ((height - constrain(mouseY, 50, height - 50)) - 50) / (float)(height - 100) * 210;
        break;
      case 2:
        gewicht = ((height - constrain(mouseY, 50, height - 50)) - 50) / (float)(height - 100) * 140;
        break;
    }
  }else{
    selectedSlider = 0;
  }
  
  drawSlider("Lengte", nf(lengte, 1, 1), width - 150, 50, 50, height - 100, lengte / 210);
  drawSlider("Gewicht", nf(gewicht, 1, 1), width - 75, 50, 50, height - 100, gewicht / 140);
}

void drawSlider(String name, String text, int x, int y, int sWidth, int sHeight, float per){
  fill(#CB4646);
  stroke(#670909);
  rect(x, y, sWidth, sHeight);
  
  fill(#670909);
  stroke(#CB4646);
  rect(x + 1, y + (sHeight - 21) - (per  * (sHeight - 21)), sWidth - 2, 20);
  
  fill(#CB4646);
  stroke(#670909);
  text(text, (x + sWidth / 2) - textWidth(text) / 2, y + (sHeight - 21) - (per  * (sHeight - 21)) + 14);
  text(name, (x + sWidth / 2) - textWidth(name) / 2, y - 4);
}

boolean mouseInSlider(int x, int y, int sWidth, int sHeight, float per){
  if(mouseX < x){
    return false;
  }
  
  if(mouseX > x + sWidth){
    return false;
  }
  
  if(mouseY < y + (sHeight - 21) - (per * (sHeight - 21))){
    return false;
  }
  
  if(mouseY > y + (sHeight - 21) - (per * (sHeight - 21)) + 20){
    return false;
  }
  
  return true;
}

Chuck a Luck - Analyse

Welke functies moet je bouwen om dit spel te spelen.

Uitgangspunt: een functie heeft één taak.

  • functie dobbel() met uitkomst 1,2 of 3
    • willekeurig getal tussen 1 en 6
  • functie winstfactor() met uitkomst -1, 1, 2, 10
    • tel aantal dobbelstenen dat geluksnummer toont
  • functie speel() die dobbel() en winstfactor() gebruikt om winst te berekenen
