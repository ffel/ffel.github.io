void tekenGezinnen(int aantalGezinnen,int imgBreedte, int imgHoogte){
  int windowW;
  int windowH;
  PImage jongetje;
  PImage meisje;
  jongetje = loadImage("jongetje.png");
  meisje = loadImage("meisje.png");
  // 1 = jongen, 2 = meisje
  int randomGezin = ceil(random(0,2));
  int xmarge = 0;
  int ymarge = 0;
  windowW = 300;
  windowH = aantalGezinnen * imgHoogte;
  size(windowW,windowH);
  int teller = 1;
  while(teller <= aantalGezinnen){
      while(randomGezin == 2){
        image(meisje,xmarge,ymarge);
        xmarge = xmarge + imgBreedte;
        randomGezin = ceil(random(0,2));
      }
      while(randomGezin ==1 && teller <= aantalGezinnen){
        image(jongetje,xmarge,ymarge);
        xmarge = 0;
        ymarge = ymarge +imgHoogte;
        teller ++;
        randomGezin = ceil(random(0,2));
      }
  }
}

Tot slot (3)

PImage jongenSprite = loadImage("http://i.imgur.com/4LvCGmN.png");
PImage meisjeSprite = loadImage("http://i.imgur.com/0NfpQnb.png");

int gezinnen = 8;
int maxGezinsgrootte = 6;
int figuurBreedte = max(jongenSprite.width, meisjeSprite.width);
int figuurHoogte = max(jongenSprite.height, meisjeSprite.height);

int schermW = maxGezinsgrootte * figuurBreedte;
int schermH = gezinnen * figuurHoogte;

int JONGEN = 0;
int MEISJE = 1;

void setup () {
  size(schermW, schermH);

  for (int i = 0; i < gezinnen; i++)
    genereerGezin( i * figuurHoogte );
}

void genereerGezin ( int y ) {
  for (int p = 0; p < maxGezinsgrootte; p++){
    int kind = genereerKind();
    tekenFiguur(kind, p * figuurBreedte, y);
    if (kind == JONGEN)
      break;
  }
}

int genereerKind () {
  if ( random(0,1) > 0.5 )
    return JONGEN;
  else
    return MEISJE;
}

void tekenFiguur ( int figuur, int x, int y ) {
  if (figuur == JONGEN)
    image ( jongenSprite, x, y);
  else
    image ( meisjeSprite, x, y);
}
